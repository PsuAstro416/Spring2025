Question:

Is there another type of plot to show the density of data that is not a contour plot?

let
    m = 40
    z = randn(n,m)
    z = (1:n)' * 1 .+ 0.5.* randn(m)
    heatmap(1:n, 1:m, z,
    c=:diverging_bwr_20_95_c54_n256,
    xlabel="x", ylabel="y",
    title="Heatmap")
end
